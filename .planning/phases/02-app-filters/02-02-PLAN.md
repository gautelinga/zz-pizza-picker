    ### `<script lang="ts">` block — write EXACTLY this logic (do not paraphrase, do not add extras):

    ```typescript
    import type { Pizza, Filter } from '../types/pizza';

    let { pizzas }: { pizzas: Pizza[] } = $props();

    let activeFilter = $state<Filter>('all');
    let result = $state<Pizza | null>(null);
    let spinning = $state(false);

    let filteredPizzas = $derived.by(() => {
      if (activeFilter === 'veg') return pizzas.filter((p) => p.vegetarian);
      if (activeFilter === 'meat') return pizzas.filter((p) => !p.vegetarian);
      return pizzas;
    });

    // Clear any displayed result when the filter changes (RESEARCH.md Pitfall 5).
    $effect(() => {
      activeFilter; // track filter as a dependency
      result = null;
    });

    function spin() {
      if (spinning) return;                       // guard re-entry (Pitfall 3)
      if (filteredPizzas.length === 0) return;    // FILT-03 safety
      spinning = true;
      result = null;
      setTimeout(() => {
        const idx = Math.floor(Math.random() * filteredPizzas.length);
        result = filteredPizzas[idx];
        spinning = false;
      }, 1500);
    }

    const filters: { id: Filter; label: string }[] = [
      { id: 'all', label: 'All' },
      { id: 'veg', label: 'Veg' },
      { id: 'meat', label: 'Meat' },
    ];
    ```

<threat_model>
## Trust Boundaries

| Boundary | Description |
|----------|-------------|
| build-time menu.json → rendered HTML + hydration payload | Scraped pizza data (name, description, price) is embedded into static HTML and into the client-side hydration state. If the scraper output were ever tampered with, strings could contain HTML/script markup. |
| client UI state → DOM | Filter state and spin results are bound into the DOM via Svelte's reactive rendering. |
| npm supply chain → compiled bundle | `@astrojs/svelte`, `svelte`, `tailwindcss`, `@tailwindcss/vite` compile user code into the shipped bundle. |

## STRIDE Threat Register

| Threat ID | Category | Component | Disposition | Mitigation Plan |
